<?php 

// This sample will show all SYSTEM authentication information objects.

if ( count($argv) == 1 ) {
  echo "Please pass me the name of a queuemanager as argument", PHP_EOL;
  exit; 
}

$json = array('AuthInfoName' => 'SYSTEM*');
$content = json_encode($json);
  
$curl = curl_init('http://localhost:8081/api/authinfo/inquire/' . $argv[1]);
curl_setopt($curl, CURLOPT_CUSTOMREQUEST, "POST");
curl_setopt($curl, CURLOPT_POSTFIELDS, $content);
curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);
curl_setopt($curl, CURLOPT_HTTPHEADER, 
  array( 
      'Content-Type: application/json',
      'Content-Length: ' . strlen($content)
  )
);

if ( ($response = curl_exec($curl)) === false ) {
  $err = curl_error($curl);
  echo "An HTTP error occurred while inquiring authentication information:"
    , $err
    , PHP_EOL;
}
else {
  $json = json_decode($response);
  if ( isset($json->error) ) {
    echo "An MQ error occurred while inquiring authentication information objects."
      , PHP_EOL;
    echo "Reason Code: "
      , $json->error->reason->code
      , " - "
      , $json->error->reason->desc
      , PHP_EOL;
  }
  else {
    if ( isset($json->authinfos) && count($json->authinfos) > 0 ) {
      foreach($json->authinfos as $authinfo)
      {
        echo $authinfo->AuthInfoName->value
          , PHP_EOL;
      }
    }
    else
    {
      echo "No authentication information objects found"
        , PHP_EOL;
    }
  }
}
